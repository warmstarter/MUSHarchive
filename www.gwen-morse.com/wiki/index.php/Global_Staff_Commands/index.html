Global Staff Commands

From Gwen Morse's Wiki
Jump to: navigation, search

This came from an old World of Darkness game set in Colorado. 

@create Global Staff Commands............GSC=10
@Desc Global Staff Commands............GSC=Contents (via +scan):%r[u(#225/FN_SCAN,%!)]
&CMD_JOIN Global Staff Commands............GSC=$+join *:@pemit *%#=You will be joining [capstr(fullname(num(*%0)))] in a few seconds.;@pemit/contents [loc(*%0)]=\[STAFF/JOIN\] [name(%#)] will be joining you in a few seconds.;@wait 2=@tel *%#=[loc(*%0)]
&CMD_SUMMON Global Staff Commands............GSC=$+summon *:@pemit %#=You summon [capstr(fullname(num(*%0)))] to you.;@wait 3=@pemit *%0=\[STAFF/SUMMON\] You suddenly feel yourself suddenly pulled in a different direction...then..POOF!;@pemit/contents loc(*%0)=\[STAFF/SUMMON\] [fullname(num(*%0))] glances around, looking confused before suddenly disappearing in a puff of smoke.;&last-loc *%0=loc(*%0);@wait 5=@tel *%0=loc(%#)
&ISSTAFF Global Staff Commands............GSC=[orflags(%#,wWiZ)]
@lock/UseLock Global Staff Commands............GSC=ISSTAFF/1
&DO_BUSY Global Staff Commands............GSC=$+busy:&status %#=busy;@pemit %#=You set yourself busy.
&DO_UNBUSY Global Staff Commands............GSC=$+unbusy:&status %#=unbusy;@pemit %#=You set yourself unbusy.
&CMD_RETURN Global Staff Commands............GSC=$+return *:@switch [hasattr(*%0,last-loc)]=0,{@pemit %#=Person has already been returned or has no last location!},{setq(0,get_eval(*%0/last-loc));@pemit *%0=\[STAFF RETURN\] You suddenly feel yourself suddenly pulled in a different direction...then..POOF!...you are back in your original location.;@pemit/contents [loc(*%0)]=\[STAFF RETURN\] [name(*%0)] is sent back to wence they came.;@pemit/contents [%q0]=\[STAFF RETURN\] [name(*%0)] suddenly appears in a puff of smoke.;@tel *%0=[get_eval(*%0/last-loc)];&last-loc *%0}
&CMD_WALL Global Staff Commands............GSC=$+wall *:@dolist [lwho()]=@pemit ##=Announcement: [switch(%0,:*,[name(%#)]%b[after(%0,:)],\;*,[name(%#)][after(%0,\;)],~*,[after(% 0,~)],[name(%#)]%bsays "%0")]
&DO_SET_QUOTA Global Staff Commands............GSC=$+setquota *=*:@pemit %#=[setq(9,pdbref(%0))][switch(%q9,#-1 NO MATCH,{'%0' is not a valid player.},{switch(isnum(%1),0,{'%1' must be a number.},{Setting quota for [name(%q9)] to %1.})})]
&DO_LASTSTAFF Global Staff Commands............GSC=$+laststaff:@pemit %#=[u(GMD/header,The last login times for staff members)];@pemit %#=[list(get(GMC/staff_list),[setq(9,get(##/last))][setq(8,sub(secs(),convtime(%q9)))][name(##)][space(sub(25,strlen(name(##))))]%q9 \([singletime(%q8)] ago\)%cn)]
&DO_SETADMIN Global Staff Commands............GSC=$+setadmin *:[setq(0,pdbref(%0))];@switch [isdbref(%q0)]=0,{@pemit %#=Sorry, '%0' is not a valid player.},{@pemit %#=Setting [name(%q0)] (%q0) up as an admin.; @set %q0=STAFF;@set %q0=ROYALTY;@power %q0=BUILDER;@power %q0=ANNOUNCE;@power %q0=COMM_ALL;@power %q0=EXPANDED_WHO;@power %q0=FIND_UNFINDABLE;@power %q0=FREE_QUOTA;@power %q0=FREE_MONEY;@power %q0=HIDE; @power %q0=MONITOR; @force %q0=addcom st=Staff;@force %q0=addcom ss=StaffSpam;@force %q0=addcom g=Guests;@switch [match(get(GMC/STAFF_LIST),%q0)]=0,{[setq(1,get(GMC/STAFF_LIST))]; &STAFF_LIST GMC=%q1 %q0}; @switch [hasattr(%q0,BIT)]=0,{&BIT %q0=:%cmAdmin%cn:}}
&DO_SHEET_1 Global Staff Commands............GSC=$+sheet *:[setq(9,pdbref(%0))]; @switch [isdbref(%q9)]=0,{@pemit %#=Sorry, '%0' is not a valid player.},{@pemit %#=[u(GMD/HEADER_LINE,Character Sheet For [name(%q9)])]; [setq(0,ulocal(CGF/FN_NOHIDDEN,u(CGF/SHEET-FMT)))]; @dolist/delimit | %q0={@pemit %#=%b%b[u(CGF/SHOW_SECTION_HEADER,##,76)];[u(GMD/FN_MAKE_COLUMNS,u(CGF/SHOW_SECTION_BODY,%q9,##),~,|,76,X:X)]}; @force %#=@pemit %#=[u(GMD/HEADER_LINE,End: Character Sheet For [name(%q9)])]}
&DO_DELADMIN Global Staff Commands............GSC=$+deladmin *:[setq(0,pdbref(%0))];@switch [isdbref(%q0)]=0,{@pemit %#=Sorry, '%0' is not a valid player.},{@pemit %#=Removing [name(%q0)]'s (%q0) admin status.; @set %q0=!STAFF;@set %q0=!ROYALTY;@power %q0=!BUILDER;@power %q0=!ANNOUNCE;@power %q0=!COMM_ALL;@power %q0=!EXPANDED_WHO;@power %q0=!FIND_UNFINDABLE;@power %q0=!FREE_QUOTA;@power %q0=!FREE_MONEY;@power %q0=!HIDE; @power %q0=!MONITOR; @force %q0=delcom st;@force %q0=delcom ss;@force %q0=delcom g;[setq(1,get(GMC/STAFF_LIST))]; [setq(2,match(%q1,%q0))]; @switch [gt(%q2,0)]=1,{&STAFF_LIST GMC=[ldelete(%q1,%q2)]}; &BIT %q0}
@Aconnect Global Staff Commands............GSC=@switch [isstaff(%#)]=1,{@switch [hasflag(%#,DEAD)]=1,@pemit %#=%rWelcome back.%r;@set %#=!DEAD}
&FLOOK Global Staff Commands............GSC=$+flook *:@pemit %#=[switch(%0,here,get_eval(loc(%#)/faedesc),get_eval(%0/faedesc))]
&STAFFALT_ADD Global Staff Commands............GSC=$+staffalt/add *:[setq(0,pdbref(%0))];@switch %q0=#-1 NO MATCH,{@pemit %#=Sorry, %0 is not a valid player},{[setq(1,v(alts_%#))][setq(3,setunion(%q1,%q0))];&alts_%# me=%q3;[setq(4,v(alts_%#))];@pemit %#=New alt list: [iter(%q4,[name(##)])]}
&ALTS_#7 Global Staff Commands............GSC=#254 #995
&STAFFALT_LIST1 Global Staff Commands............GSC=$+staffalt/list:[setq(1,v(alts_%#))];@pemit %#=List of your alts: [iter(%q1,name(##))]
&STAFFALT_LIST2 Global Staff Commands............GSC=$+staffalt/list *:[setq(0,pdbref(%0))];@switch %q0=#-1 NO MATCH,{@pemit %#=Sorry, %0 is not a valid player},{[setq(1,v(alts_%q0))];@pemit %#=Alt list for [name(%q0)]: [iter(%q1,name(##))]}
&STAFFALT_LIST3 Global Staff Commands............GSC=$+staffalt/listall:@pemit %#=[u(GMD/header,List of Staff Alts)];[list(u(GMC/STAFF_LIST),[setq(1,v(alts_##))]Alt list for [name(##)]: [u(STAFFALT_LIST,%q1)])]
&STAFFALT_LIST Global Staff Commands............GSC=[iter(%0,name(##))]
&STAFFALT_DEL Global Staff Commands............GSC=$+staffalt/del *:[setq(0,pdbref(%0))];@switch %q0=#-1 NO MATCH,{@pemit %#=Sorry, %0 is not a valid player},{[setq(1,v(alts_%#))][setq(2,match(%q1,%q0))][setq(3,ldelete(%q1,%q2))];&alts_%# me=%q3;[setq(4,v(alts_%#))];@pemit %#=New alt list: [iter(%q4,[name(##)])]}
&DO_SETADMIN1 Global Staff Commands............GSC=$+setadmin1 *:[setq(0,pdbref(%0))];@switch [isdbref(%q0)]=0,{@pemit %#=Sorry, '%0' is not a valid player.},{@pemit %#=Setting [name(%q0)] (%q0) up as an admin.; @set %q0=STAFF;@set %q0=ROYALTY;@power %q0=BUILDER;@power %q0=ANNOUNCE;@power %q0=COMM_ALL;@power %q0=EXPANDED_WHO;@power %q0=FIND_UNFINDABLE;@power %q0=FREE_QUOTA;@power %q0=FREE_MONEY;@power %q0=HIDE;@power %q0=MONITOR}
&ALTS_#4 Global Staff Commands............GSC=#8
&ALTS_#812 Global Staff Commands............GSC= #932
&DO_APPROVE Global Staff Commands............GSC=$+approve *:[setq(9,locate(%#,%0,*))]; @switch isdbref(%q9)=1,{[setq(2,time())]; @switch [strmatch(PLAYER,type(%q9))]=1,{@pemit %#=This command is not for players, use +approve/rp %0 instead.},{@set %q9=UNINSPECTED; &APPROVAL %q9=Approved on %q2 by %n.; @pemit %#=You have approved [name(%q9)] dated %q2.}},{@pemit %#=Invalid object.}
&ALTS_#36 Global Staff Commands............GSC= #229
&DO_DELADMIN1 Global Staff Commands............GSC=$+deladmin1 *:[setq(0,pdbref(%0))];@switch [isdbref(%q0)]=0,{@pemit %#=Sorry, '%0' is not a valid player.},{@pemit %#=Removing [name(%q0)]'s (%q0) admin status.; @set %q0=!STAFF;@set %q0=!ROYALTY;@power %q0=!BUILDER;@power %q0=!ANNOUNCE;@power %q0=!COMM_ALL;@power %q0=!EXPANDED_WHO;@power %q0=!FIND_UNFINDABLE;@power %q0=!FREE_QUOTA;@power %q0=!FREE_MONEY;@power %q0=!HIDE; @power %q0=!MONITOR}
@set Global Staff Commands............GSC=INHERIT
@set Global Staff Commands............GSC=SAFE
@set Global Staff Commands............GSC=STICKY
