Good Enough AnitaVerse Chargen

Original source: http://softcode.livejournal.com/683.html 

@dig/teleport GoodEnough CharGen

@Desc here=[parse(lnum(inc(abs(get(%#/_STEP)))), u(DESC-##))]

&DESC-0 here=%RSTEP 0> [if(hasattr(%#, _STEP), +restart to clear the slate., +start to set things up.)]%R

&DESC-1 here=%RSTEP 1> [if(hasattr(%#, _TRAITS), Your Traits are: [itemize(get(%#/_TRAITS),|)], {Order traits with '+traits . Traits are Mental, Physical, Social, and Psyche. The first one listed will be lowest, the last one highest. EG: '+traits [shuffle(Physical Mental Social Psyche)]'})]%R

&DESC-2 here=%RSTEP 2> You have [if(words(get(%#/_SKILLS)), itemize(parse(sort(setunion(get(%#/_SKILLS),), a), [if(dec(words(matchall(get(%#/_SKILLS), ##))), Advanced, Basic)] ##,,|),|), none)]. [if(lt(words(get(%#/_SKILLS)), u(SKILL-PTS)), {You have [sub(u(SKILL-PTS), words(get(%#/_SKILLS)))] points for Skills. Skills are [itemize(u(SKILLS))]. Basic costs one point, Advanced costs two points. You can buy them with '+skill '. EG: '+skill [pickrand(u(SKILLS))]'})]%R

&DESC-3 here=%RSTEP 3> [if(hasattr(%#, _RACE), You are a [get(%#/_RACE)]-type., Pick race! +race can be one of [itemize(u(RACES),, or)]. EG: '+race [pickrand(u(RACES))]')]%R

&DESC-4 here=%RSTEP 4> Type '+finished' if finished. %R

&RESTART-CMD here=$+*start:&_STEP %#=1 ;@pemit %#=CHARGEN> You clean the slate. ;@dolist _FINISHED _RACE _TRAITS _SKILLS=@wipe %#/## ;@force %#=LOOK

&TRAIT-CMD here=$+traits *:@switch/first 1= not(match(get(%#/_STEP), 1)), {@pemit %#=CHARGEN> You're not at the trait stage. Please +restart if you want to clear it all. }, hasattr(%#, _FINISHED), {@pemit %#=CHARGEN> You're done. Please +restart if you want to clear it all. }, lor(parse(setr(0, lcstr(secure(%0))), not(isword(##)))), {@pemit %#=CHARGEN> Please enter just the names of the traits. EG: '+traits [shuffle(Physical Mental Social Psyche)]' }, not(land(parse(%q0, strmatch(physical mental social psyche, *##*)))), {@pemit %#=CHARGEN> Please enter the names of the traits. EG: '+traits [shuffle(Physical Mental Social Psyche)]' }, abs(comp(4, words(setr(1, setinter(mental physical psyche social, sort(parse(%q0, grab(mental physical psyche social, ##*)))))))) , {@pemit %#=CHARGEN> Please enter all four traits. EG: '+traits [shuffle(Physical Mental Social Psyche)]' }, {@pemit %#=CHARGEN> You set your traits to [parse(extract(%q0, 1, 4), [capstr(grab(mental physical psyche social, ##*))]:#@ )] ;&_TRAITS %#=[parse(extract(%q0, 1, 4), [grab(mental physical psyche social, ##*)]:#@,,|)] ;&_STEP %#=2 ;@force %#=LOOK }

&SKILL-CMD here=$+skill *:@switch/first 1= not(match(get(%#/_STEP), 2)), {@pemit %#=CHARGEN> You're not at the skill stage. Please +restart if you want to clear it all. }, hasattr(%#, _FINISHED), {@pemit %#=CHARGEN> You're done. Please +restart if you want to clear it all. }, gte(words(get(%#/_SKILLS)), u(SKILL-PTS)), {@pemit %#=CHARGEN> You're out of skill points.}, not(isword(trim(secure(%0)))), {@pemit %#=CHARGEN> Please enter just the name of the skill. EG: '+skill [first(shuffle(u(SKILLS)))]' }, gt(words(secure(%0)), 1), {@pemit %#=CHARGEN> Just one skill at a time, please. }, not(words(setr(0, grab(u(SKILLS), [secure(%0)]*)))), {@pemit %#=CHARGEN> '[secure(%0)]' is not a skill. Try one of [itemize(u(SKILLS))] }, gte(words(matchall(get(%#/_SKILLS), %q0)), 2), {@pemit %#=CHARGEN> You already have Advanced %q0. That's as good as it gets.}, {@pemit %#=CHARGEN> You take [if(match(get(%#/_SKILLS), %q0), an Advanced, a Basic)] level of the %q0 Skill. ;&_SKILLS %#=setr(1, cat(get(%#/_SKILLS), %q0)) ;@if gte(words(%q1), u(SKILL-PTS))= {@pemit %#=CHARGEN> Skills done! ;&_STEP %#=3 ;@force %#=look }, {@pemit %#=CHARGEN> Keep +skilling! You have [sub(u(SKILL-PTS), words(%q1))] point[if(dec(sub(u(SKILL-PTS), words(%q1))),,s)] left. } }

&RACE-CMD here=$+race *:@switch/first 1= not(match(get(%#/_STEP), 3)), {@pemit %#=CHARGEN> You're not at the race stage. Please +restart if you want to clear it all. }, hasattr(%#, _FINISHED), {@pemit %#=CHARGEN> You're done. Please +restart if you want to clear it all. }, hasattr(%#, _RACE), {@pemit %#=CHARGEN> You're already a [get(%#/_RACE)]-type. Please +restart if you want to clear it all. }, not(words(setr(1, grab(u(RACES), *[secure(%0)]*)))), {@pemit %#=CHARGEN> Please '+race [type]'. With [type] being one of [itemize(u(RACES),, or)]. }, {@pemit %#=CHARGEN> You set yourself up as a %q1-type. ;&_RACE %#=%q1 ;@if hasattr(me, BOOST-%q1)= {@dolist u(BOOST-%q1)=&_TRAITS %#=[replace(setr(0, get(%#/_TRAITS)), setr(2, match(%q0, ##:*, |)), ##:[inc(after(extract(%q0, %q2, 1, |), :))], |)] } ;&_STEP %#=4 ;@wait 0=@force %#=look }

&SKILLS here=Hide Seek Hit Run Know Talk Make Fake

&SKILL-PTS here=5

&BOOST-VAMPIRE here=mental physical physical psyche

&BOOST-SHAPESHIFTER here=social physical physical psyche

&BOOST-MAGIC here=psyche

&RACES here=human vampire shapeshifter magic

&FINISHED-CMD here=$+finish*:@if eq(get(%#/_STEP), 4)= {@pemit %#=CHARGEN> Good for you! ;&_FINISHED %#=[time()] ;&_STEP %#=0 ;&_SKILLS %#=parse(setunion(setr(0, sort(get(%#/_SKILLS), a)),), ##:[words(matchall(%q0, ##))],,|) }, {@pemit %#=CHARGEN> I think not. ;@force %#=LOOK }

@set here=INHERIT
