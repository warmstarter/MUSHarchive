@create Shifter FormChange Global <SFG>=10
@lock Shifter FormChange Global <SFG>=#2
&CMD_+SHIFT Shifter FormChange Global <SFG>=$+shift *:@switch/first [strmatch(get(%#/race),were)][gt(match(v(list_forms),%0),0)][not(strmatch(get(%#/current_form),%0))][gt(match(extract(v(list_forms),u(fn_which_list,%#),1,|),%0),0)]=0*,{@pemit %#=Huh?%b%b(Type "help" for help.)},10*,{@pemit %#=That form is not anywhere within the system.},110*,{@pemit %#=You are already in that form.},1110,{@pemit %#=That form does not match the type of shifter you are.},1111,{&current_form %#=[capstr(%0)];&tattr %#=[setq(0,0)][iter(get(%#/attr),[setq(0,add(%q0,1))][setq(1,add(##,[setq(2,extract(v([switch(get(%#/class),bastet,bastet_[get(%#/tribe)]_%0,mokole,mokole_[get(%#/tribe)]_%0,[get(%#/class)]_%0)]),%q0,1))][switch(%q2,X,-##,%q2)]))][switch(lt(%q1,0),1,0,%q1)])] ;@desc %#=[get(%#/desc_%0)];@tr %#/ashift_%0;@remit %l=%N [switch(hasattr(%#,shift_%0),1,[get(%#/shift_%0)],shifts into %p %0 form.)];@pemit %#=You shifted to "[capstr(%0)]".}
&CMD_+SHIFT_LIST Shifter FormChange Global <SFG>=$+shift/list:@switch [strmatch(get(%#/race),were)]=1,{@pemit %#=[center(*%ch%cr +Shift/List Report %cn*,79,-)][ljust(%b%b%bRace:,20)]Forms:%r[u(#560/drawline)]%r%b[setq(0,0)][iter(Garou Nuwisha Gurahl Corax Ratkin Mokole Rokea Ananasi Bastet,[ljust(%b%b##,15)] - [setq(0,add(%q0,1))][extract(v(list_forms),%q0,1,|)]%r)][repeat(-,79)]%r},{@pemit %#=Huh? &DO_GNOSIS Shifter FormChange Global <SFG>=$+recover gnosis=*:[setq(8,abs(%0))]; @switch/first 0 = gte(u(%#/gnosis),1),{@pemit %# = Huh? \(Type "help" for help.\)},isnum(%0),{@pemit %# = CODE: '%0' is not a proper numeric value for the number of hours you would like to meditate to regain Gnosis.},not(match(get(%#/gnosis),get(%#/gps))),{@pemit %# = CODE: You have as much gnosis as you can hold.},not(hasattr(me,valid_%#)),{@pemit %# = CODE: You are already meditating to regain Gnosis.},{[setq(5,rolls(add(stat(%#,Perception),stat(%#,Meditation)),8))] [setq(6,sub(u(%#/gnosis),u(%#/gps)))] [setq(7,abs(%0))] [setq(8,min(%q5,%q6,%q7))]; @pemit %# = CODE: The results of your meditation attempt to regain Gnosis will be provided to you in '%q7' RL hour(s). You now have [get(%#/rp)] point(s).; &LASTRAGE %#=vsecs(); @wait 0=@lock %#/LASTRAGE},{@pemit %#=Bete cannot yet recover rage.}}
